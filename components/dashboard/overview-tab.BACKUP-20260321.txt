"use client"

import { useState, useMemo } from "react"
import { Card, CardContent, CardDescription, CardHeader, CardTitle } from "@/components/ui/card"
import { CalendarHeatmap } from "./calendar-heatmap"
import { PlatformBreakdownPie } from "@/components/dashboard/platform-breakdown-pie"
import { TopContentTab } from "./top-content-tab"
import { PostDrilldown } from "./post-drilldown"
import { WeeklySummaryHeader } from "./weekly-summary-header"
import { 
  BarChart3, Eye, MousePointerClick, TrendingUp, Users, UserPlus,
  Globe, Target, ArrowUpRight, ArrowDownRight, Minus
} from "lucide-react"
import { 
  AreaChart, Area, BarChart, Bar, 
  ResponsiveContainer, Tooltip as RechartsTooltip, XAxis
} from "recharts"
import type { ParsedDataset, DataPoint } from "@/lib/parsers/types"
import type { GAUTMCampaign } from "@/lib/parsers/google-analytics-parser"

interface OverviewTabProps {
  data?: ParsedDataset | null
  platform: string
  dateRange: string
  webData?: ParsedDataset | null
}

function normalizePlatform(platform: string): string {
  const normalized = platform.toLowerCase()
  if (normalized === 'x') return 'twitter'
  return normalized
}

function formatNumber(num: number): string {
  if (num >= 1000000) return `${(num / 1000000).toFixed(1)}M`
  if (num >= 1000) return `${(num / 1000).toFixed(1)}K`
  return Math.round(num).toString()
}

function getGreeting(): string {
  const hour = new Date().getHours()
  if (hour < 12) return "Good Morning"
  if (hour < 18) return "Good Afternoon"
  return "Good Evening"
}

/**
 * Mini KPI card with optional sparkline
 */
function OverviewKPICard({ 
  label, value, subText, icon, trend, sparklineData, sparklineColor 
}: { 
  label: string
  value: string | number
  subText: string
  icon: React.ReactNode
  trend?: 'up' | 'down' | 'neutral'
  sparklineData?: { value: number }[]
  sparklineColor?: string
}) {
  return (
    <Card className="relative overflow-hidden">
      <CardContent className="pt-5 pb-4 px-5">
        <div className="flex items-start justify-between mb-1">
          <p className="text-sm font-medium text-neutral-500">{label}</p>
          <div className="p-2 rounded-lg bg-neutral-100 dark:bg-neutral-800">
            {icon}
          </div>
        </div>
        <div className="flex items-end justify-between">
          <div>
            <p className="text-2xl font-bold text-foreground tracking-tight">{value}</p>
            <div className="flex items-center gap-1 mt-1">
              {trend === 'up' && <ArrowUpRight className="w-3.5 h-3.5 text-green-500" />}
              {trend === 'down' && <ArrowDownRight className="w-3.5 h-3.5 text-red-500" />}
              {trend === 'neutral' && <Minus className="w-3.5 h-3.5 text-neutral-400" />}
              <p className="text-xs text-neutral-500">{subText}</p>
            </div>
          </div>
          {sparklineData && sparklineData.length > 1 && (
            <div className="w-24 h-10">
              <ResponsiveContainer width="100%" height="100%">
                <AreaChart data={sparklineData}>
                  <defs>
                    <linearGradient id={`grad-${label.replace(/\s/g, '')}`} x1="0" y1="0" x2="0" y2="1">
                      <stop offset="0%" stopColor={sparklineColor || '#3b82f6'} stopOpacity={0.3} />
                      <stop offset="100%" stopColor={sparklineColor || '#3b82f6'} stopOpacity={0} />
                    </linearGradient>
                  </defs>
                  <Area 
                    type="monotone" 
                    dataKey="value" 
                    stroke={sparklineColor || '#3b82f6'} 
                    strokeWidth={1.5}
                    fill={`url(#grad-${label.replace(/\s/g, '')})`}
                    dot={false}
                    isAnimationActive={false}
                  />
                </AreaChart>
              </ResponsiveContainer>
            </div>
          )}
        </div>
      </CardContent>
    </Card>
  )
}

const MEDIUM_COLORS: Record<string, string> = {
  'organic': '#22c55e',
  'direct': '#3b82f6',
  '(none)': '#3b82f6',
  'referral': '#a855f7',
  'website': '#8b5cf6',
  'post': '#0ea5e9',
  'email': '#f97316',
  '(not set)': '#94a3b8',
}

export function OverviewTab({ data, platform, dateRange, webData }: OverviewTabProps) {
  const [drilldownOpen, setDrilldownOpen] = useState(false)
  const [selectedDate, setSelectedDate] = useState<string | null>(null)
  const [selectedPosts, setSelectedPosts] = useState<DataPoint[]>([])

  // === SOCIAL KPIs ===
  const contentKPIs = useMemo(() => {
    if (!data) return null
    
    let filteredPoints = data.dataPoints
    if (platform !== "All") {
      const normalizedFilter = normalizePlatform(platform)
      filteredPoints = filteredPoints.filter(p => p.source.toLowerCase() === normalizedFilter)
    }
    
    if (filteredPoints.length === 0) return null
    
    const totals = filteredPoints.reduce((acc, point) => ({
      totalImpressions: acc.totalImpressions + (Number(point.metrics.impressions) || 0),
      totalEngagements: acc.totalEngagements + (Number(point.metrics.engagements) || 0),
      totalClicks: acc.totalClicks + (Number(point.metrics.clicks) || 0),
      totalEngagementRates: acc.totalEngagementRates + (Number(point.metrics.engagement_rate) || 0),
    }), { totalImpressions: 0, totalEngagements: 0, totalClicks: 0, totalEngagementRates: 0 })
    
    return {
      totalPosts: filteredPoints.length,
      totalImpressions: totals.totalImpressions,
      totalEngagements: totals.totalEngagements,
      totalClicks: totals.totalClicks,
      avgEngagementRate: totals.totalEngagementRates / filteredPoints.length,
    }
  }, [data, platform])

  // Sparkline data: daily impressions for mini chart
  const impressionsSparkline = useMemo(() => {
    if (!data) return []
    const byDate = new Map<string, number>()
    data.dataPoints.forEach(p => {
      const existing = byDate.get(p.date) || 0
      byDate.set(p.date, existing + (Number(p.metrics.impressions) || 0))
    })
    return Array.from(byDate.entries())
      .sort((a, b) => a[0].localeCompare(b[0]))
      .map(([, value]) => ({ value }))
  }, [data])

  // Sparkline data: daily engagements
  const engagementsSparkline = useMemo(() => {
    if (!data) return []
    const byDate = new Map<string, number>()
    data.dataPoints.forEach(p => {
      const existing = byDate.get(p.date) || 0
      byDate.set(p.date, existing + (Number(p.metrics.engagements) || 0))
    })
    return Array.from(byDate.entries())
      .sort((a, b) => a[0].localeCompare(b[0]))
      .map(([, value]) => ({ value }))
  }, [data])

  // === WEB ANALYTICS INSIGHTS ===
  const webInsights = useMemo(() => {
    if (!webData?.metadata?.utmCampaigns) return null
    const campaigns = webData.metadata.utmCampaigns as GAUTMCampaign[]
    if (campaigns.length === 0) return null
    
    const totalSessions = campaigns.reduce((sum, c) => sum + c.sessions, 0)
    
    // Group by medium
    const byMedium = campaigns.reduce((acc, c) => {
      let medium = c.medium || '(not set)'
      if (medium === '(none)') medium = 'direct'
      acc[medium] = (acc[medium] || 0) + c.sessions
      return acc
    }, {} as Record<string, number>)
    
    const mediumData = Object.entries(byMedium)
      .map(([medium, sessions]) => ({
        name: medium.charAt(0).toUpperCase() + medium.slice(1),
        sessions,
        fill: MEDIUM_COLORS[medium.toLowerCase()] || '#64748b'
      }))
      .sort((a, b) => b.sessions - a.sessions)
      .slice(0, 6)
    
    // Top sources
    const bySource = campaigns.reduce((acc, c) => {
      acc[c.source] = (acc[c.source] || 0) + c.sessions
      return acc
    }, {} as Record<string, number>)
    
    const topSources = Object.entries(bySource)
      .sort((a, b) => b[1] - a[1])
      .slice(0, 5)
      .map(([source, sessions]) => ({ source, sessions, pct: (sessions / totalSessions * 100) }))
    
    return { totalSessions, mediumData, topSources, campaignCount: campaigns.length }
  }, [webData])

  const handleDateClick = (dateStr: string, posts: DataPoint[]) => {
    setSelectedDate(dateStr)
    setSelectedPosts(posts)
    setDrilldownOpen(true)
  }

  const handleCloseDrilldown = () => {
    setDrilldownOpen(false)
    setSelectedDate(null)
    setSelectedPosts([])
  }

  return (
    <div className="space-y-6">
      {/* Greeting Header */}
      {data && (
        <WeeklySummaryHeader data={data} userName="Andy" />
      )}

      {/* === SOCIAL SECTION === */}
      {contentKPIs && (
        <>
          <div>
            <h2 className="text-lg font-semibold text-foreground mb-3">Content Performance</h2>
            <div className="grid grid-cols-1 md:grid-cols-2 lg:grid-cols-4 gap-4">
              <OverviewKPICard
                label="Total Posts"
                value={contentKPIs.totalPosts}
                subText={platform === "All" ? "Across all platforms" : platform}
                icon={<BarChart3 className="w-5 h-5 text-neutral-600 dark:text-neutral-300" />}
              />
              <OverviewKPICard
                label="Total Impressions"
                value={formatNumber(contentKPIs.totalImpressions)}
                subText="Views of your content"
                icon={<Eye className="w-5 h-5 text-neutral-600 dark:text-neutral-300" />}
                sparklineData={impressionsSparkline}
                sparklineColor="#3b82f6"
              />
              <OverviewKPICard
                label="Total Engagements"
                value={formatNumber(contentKPIs.totalEngagements)}
                subText="Likes, comments, shares"
                icon={<TrendingUp className="w-5 h-5 text-neutral-600 dark:text-neutral-300" />}
                sparklineData={engagementsSparkline}
                sparklineColor="#22c55e"
              />
              <OverviewKPICard
                label="Avg. Engagement Rate"
                value={`${contentKPIs.avgEngagementRate.toFixed(2)}%`}
                subText="Engagement / Impressions"
                icon={<MousePointerClick className="w-5 h-5 text-neutral-600 dark:text-neutral-300" />}
              />
            </div>
          </div>

          {/* Heatmap + Platform Breakdown */}
          <div className="grid grid-cols-1 lg:grid-cols-4 gap-4">
            <div className="lg:col-span-1">
              <PlatformBreakdownPie data={data} platform={platform} />
            </div>
            <div className="lg:col-span-3">
              <CalendarHeatmap 
                data={data} 
                platform={platform} 
                dateRange={dateRange}
                onDateClick={handleDateClick}
              />
            </div>
          </div>

          <TopContentTab data={data} platform={platform} />
        </>
      )}

      {/* === WEB ANALYTICS SNAPSHOT === */}
      {webInsights && (
        <div>
          <h2 className="text-lg font-semibold text-foreground mb-3">Web Analytics</h2>
          <div className="grid grid-cols-1 lg:grid-cols-3 gap-4">
            
            {/* Sessions KPI card */}
            <OverviewKPICard
              label="Total Sessions"
              value={formatNumber(webInsights.totalSessions)}
              subText={`${webInsights.campaignCount} active campaigns`}
              icon={<Globe className="w-5 h-5 text-neutral-600 dark:text-neutral-300" />}
            />

            {/* Traffic by Medium - horizontal bars */}
            <Card className="lg:col-span-2">
              <CardContent className="pt-5 pb-4 px-5">
                <div className="flex items-center justify-between mb-3">
                  <div>
                    <p className="text-sm font-medium text-neutral-500">Traffic by Medium</p>
                    <p className="text-xs text-neutral-400 mt-0.5">Session distribution</p>
                  </div>
                  <div className="p-2 rounded-lg bg-neutral-100 dark:bg-neutral-800">
                    <Target className="w-5 h-5 text-neutral-600 dark:text-neutral-300" />
                  </div>
                </div>
                <div className="space-y-2.5">
                  {webInsights.mediumData.map((medium, idx) => (
                    <div key={idx}>
                      <div className="flex items-center justify-between text-xs mb-1">
                        <span className="text-foreground font-medium">{medium.name}</span>
                        <span className="text-neutral-400">
                          {medium.sessions.toLocaleString()} ({(medium.sessions / webInsights.totalSessions * 100).toFixed(0)}%)
                        </span>
                      </div>
                      <div className="w-full bg-neutral-100 dark:bg-neutral-800 rounded-full h-1.5">
                        <div 
                          className="rounded-full h-1.5 transition-all"
                          style={{ 
                            width: `${(medium.sessions / webInsights.mediumData[0].sessions) * 100}%`,
                            backgroundColor: medium.fill
                          }}
                        />
                      </div>
                    </div>
                  ))}
                </div>
              </CardContent>
            </Card>
          </div>
        </div>
      )}

      {/* No data at all */}
      {!contentKPIs && !webInsights && (
        <div className="text-center py-12">
          <p className="text-neutral-500">No metrics available to display</p>
        </div>
      )}

      <PostDrilldown
        open={drilldownOpen}
        dateStr={selectedDate}
        posts={selectedPosts}
        onClose={handleCloseDrilldown}
      />
    </div>
  )
}
